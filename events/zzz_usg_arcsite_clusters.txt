# Unique Systems Guaranteed — per-event override of ancrel.12050.
#
# Vanilla ancrel.12050 ("Surveillance Supercomputer" spawn) rolls a 1:1:2 random_list:
#   1 = Ultima Vigilis cluster (ancrel.12055), 1 = Chosen cluster (fircon.3500), 2 = nothing.
# This override forces BOTH special cluster systems to always spawn.
#
# Only this single event is overridden (loads after vanilla via the zzz_ name), so every
# other ancient-relics arcsite event comes from current vanilla — unlike the old full-file
# copy, which froze the entire 10k-line event file at an old patch.
namespace = ancrel

event = {
	id = ancrel.12050
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		# Ultima Vigilis / Mindwardens cluster (Ancient Relics)
		random_rim_system = {
			limit = { is_mindwardens_story_system = no }
			system_event = { id = ancrel.12055 }
		}
		# Chosen cluster (First Contact) — guarded so it is a safe no-op without that DLC
		if = {
			limit = { has_first_contact_dlc = yes }
			random_rim_system = {
				limit = { is_mindwardens_story_system = no }
				system_event = { id = fircon.3500 }
			}
		}
	}
}
